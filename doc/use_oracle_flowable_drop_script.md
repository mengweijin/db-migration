    TYPE FLOWABLE_TYPE IS TABLE OF VARCHAR2 INDEX BY BINARY_INTEGER;
    -- 声明第一个数组变量 FLW_TABLE
    FLW_TABLE FLOWABLE_TYPE;
BEGIN
    -- 数组变量赋值，注意表顺序。
    FLW_TABLE(1) := 'ACT_EVT_LOG';

    -- 循环数组
    -- DBMS_OUTPUT.PUT_LINE('i=' || i || ', FLW_TABLE= ' || FLW_TABLE(i));
    FOR i IN 1 .. FLW_TABLE.COUNT LOOP
        BEGIN
            EXECUTE IMMEDIATE 'DROP TABLE ' || FLW_TABLE(i);
        EXCEPTION
            WHEN OTHERS THEN NULL;
        END;
    END LOOP;

    BEGIN
        EXECUTE IMMEDIATE 'DROP SEQUENCE ACT_EVT_LOG_SEQ';
    EXCEPTION
        WHEN OTHERS THEN NULL;
    END;

    BEGIN
        EXECUTE IMMEDIATE 'DROP SEQUENCE ACT_HI_TASK_EVT_LOG_SEQ';
    EXCEPTION
        WHEN OTHERS THEN NULL;
    END;
END;
